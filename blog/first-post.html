(setq mjl/start-time (current-time))

(defun mjl/init-log (msg)
  "Places a message to the *Messages* buffer with a date at the front"
  (message (concat (format-time-string "%Y-%m-%dT%T%z: ") msg)))

(mjl/init-log "* inhibit-startup-message")
(setq inhibit-startup-message t) ; in case it's not set elsewhere
(setq system-name-fqdn system-name) ; save original `system-name'
(setq system-name (car (split-string system-name "\\."))) ; fix FQDN

(mjl/init-log "* load package library")
(require 'package)

(mjl/init-log "* define utility init functions")

(defun mjl/init-check-install-packages (packages)
  "Check that `packages' list is installed, install any that are missing"
  (dolist (p packages)
    (when (not (package-installed-p p))
      (package-install p))))

(defun mjl/flatten (alist)
  "given `alist' with arbitrary nesting, return a new list with
   all the nesting flattened"
  (cond
   ((null alist) nil)
   ((atom alist) (list alist))
   (t
    (append (mjl/flatten (car alist)) (mjl/flatten (cdr alist))))))

  "a regex for finding elisp source code in directories")

(defvar mjl/init-files-to-load 
  (mjl/flatten 
   (list mjl/system-config
	 (when (file-exists-p mjl/system-config-dir)
           (directory-files mjl/system-config-dir t mjl/elisp-pattern))
	 mjl/user-config
	 (when (file-exists-p mjl/user-config-dir)
           (directory-files mjl/user-config-dir t mjl/elisp-pattern))
	 mjl/hooks-file
	 mjl/bindings-file))
"Ordered list of personal configuration files to load. This order
is important: system-specific setup and user configured
facilities need to be loaded first; hooks and key bindings need
to be completed last.")

(mjl/init-log "** Load system and user settings")
(mapc (lambda (f) (when (file-exists-p f) (load f)))
      mjl/init-files-to-load)
